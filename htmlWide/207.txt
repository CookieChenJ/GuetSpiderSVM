	function clear_input(){
		$("input[type!=button]").each(function(){
			$(this).val('').focus();
		})
		$("select").each(function(){
			$(this).val('').focus();
		})
	}
	function submit_form(){
		state = true;
		if($('#classname').val()==''){
			$('#classname').next().show();
			location.href='#classname';
			state = false;
			return false;
		}
		if($('#user').val()==''){
			$('#user').next().show();
			location.href='#user';
			state = false;
			return false;
		}
		if($('#department').val()==''){
			$('#department').next().show();
			location.href='#department';
			state = false;
			return false;
		}
		if($('#tel').val()==''){
			$('#tel').next().show();
			location.href='#tel';
			state = false;
			return false;
		}
		if($('#date').val()==''){
			$('#date').next().show();
			location.href='#date';
			state = false;
			return false;
		}
		if($('#time').val()==''){
			$('#time').next().show();
			location.href='#time';
			state = false;
			return false;
		}  
        
        if($('#other').val()==''){
            $('#other').next().show();
            location.href='#other';
            state = false;
            return false;
        }                    
		if(state == false){
			art.artDialog.tips('请核对相关信息');
			return false;
		}else {
			var formData = new FormData($("#applyclass")[0]);
			$.ajax({
				type: 'POST',
				data: formData,
				async: false,
				cache: false,
				dataType: 'json',
				contentType: false,
				processData: false,
				success: function (msg) {
					if (msg.state == 1) {
						Art.Location(msg.info,'/index/applyclasslist')
					}
					else {
						Art.Remark(msg.info,2,'error');
					}
				}
			});
		}
	}
</script>
