function ltrim(string) 
{ 
    var objstr = string; 

    if((objstr == "") || (objstr == " ")) 
    { 
        return ""; 
    } 

    var head; 
    var exitw = true; 

    while(exitw) 
    { 
        if((objstr == "") || (objstr == " ")) 
        { 
            return ""; 
        }

        if(head == " ") 
        { 
            objstr = objstr.substr(1); 
        } 
        else 
        { 
            exitw = false; 
        } 
    } 

    return objstr; 
}
