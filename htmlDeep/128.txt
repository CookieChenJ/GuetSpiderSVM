    {
        menu.className = i == curr ? "up" : "";
    }
        if (i == curr) {
      if(cont != null)
        {
            cont.style.display = "block";
        }
        } else {
        if(cont != null)
    {
            cont.style.display = "none";
    }
        }
    }
}
